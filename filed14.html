static boolean solvable (BinaryTree node) {

            enter(node);
            
            if (node == null) return no(node);
            
            if (node.isGoalNode) return yes(node);
            
            if (solvable(node.leftChild)) return yes(node);
            
            if (solvable(node.rightChild)) return yes(node);
            
            return no(node);
        }
