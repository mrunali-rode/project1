static boolean solvable (BinaryTree node) {

        /*1*/ if (node == null) return false;
        /*2*/ if (node.isGoalNode) return true;    
        /*3*/ if (solvable(node.leftChild)) return true;    
        /*4*/ if (solvable(node.rightChild)) return true;
        /*5*/ return false;
        
       }
